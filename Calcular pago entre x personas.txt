#include <stdio.h>
int main(void) {
    // + - * / % : arithmetic operators
    // < > <= >= != == : comparison operators

    int passengers;
    double gas;
    double cost;

    printf ("Please, submit number of passengers and price (including at least one decimal)")
    scanf ("%d %lf", &passengers, &gas); //introduce value of people to pay-1 and price to divide among people
    
    if (passengers) {
        cost = gas/(passengers+1);
        printf("Each needs to pay %.2lf €\n", cost);

    } else {
        printf("%.2lf €\n", gas);
    }

    return 0;
}

#include <stdio.h>
int main(void) {
    
    int passengers;
    double gas;
